document.addEventListener('DOMContentLoaded', () => { const input = document.querySelector('#controls input'); const createButton = document.querySelector('[data-create]'); const destroyButton = document.querySelector('[data-destroy]'); const boxesContainer = document.querySelector('#boxes'); // Генерує випадковий колір у форматі HEX function getRandomHexColor() { return `#${Math.floor(Math.random() * 16777215) .toString(16) .padStart(6, 0)}`; } // Створює та додає нові div елементи function createBoxes(amount) { const elements = []; for (let i = 0; i < amount; i++) { const size = 30 + i * 10; // Розмір збільшується на 10px для кожного наступного елемента const div = document.createElement('div'); div.style.width = `${size}px`; div.style.height = `${size}px`; div.style.backgroundColor = getRandomHexColor(); elements.push(div); } boxesContainer.append(...elements); } // Очищає контейнер function destroyBoxes() { boxesContainer.innerHTML = ''; } // Обробник натискання на кнопку Create createButton.addEventListener('click', () => { const amount = parseInt(input.value, 10); if (amount >= 1 && amount <= 100) { destroyBoxes(); // Очищує попередні елементи createBoxes(amount); // Створює нові } input.value = ''; // Очищує значення інпуту }); // Обробник натискання на кнопку Destroy destroyButton.addEventListener('click', destroyBoxes); });
